import os
import json
import logging
from typing import Dict, List, Tuple, Optional, Any
from openai import OpenAI

# Configure logging
logger = logging.getLogger(__name__)

# Get OpenAI API key
api_key = os.environ.get("OPENAI_API_KEY")
if not api_key:
    logger.warning("OpenAI API key not found in environment variables")

# Initialize OpenAI client
client = OpenAI(api_key=api_key)

# Default story options
STORY_OPTIONS = {
    "conflicts": [
        ("💣", "Double agent exposed"), ("💼", "Corporate espionage"), ("🧪", "Bioweapon heist"),
        ("💰", "Trillion-dollar ransom"), ("🔍", "Assassination conspiracy"),
        ("🕵️", "Government overthrow"), ("🌌", "Space station takeover"), ("🧠", "Mind control experiment")
    ],
    "settings": [
        ("🗼", "DC Offices"), ("🏝️", "Private Luxury Island"), ("🏙️", "Dubai Mega-Skyscraper"),
        ("🚢", "Orbital Cruise Liner"), ("❄️", "Arctic Research Base"), ("🏰", "Monaco Casino"),
        ("🏜️", "Sahara Desert Compound"), ("🌋", "Volcanic Lair")
    ],
    "narrative_styles": [
        ("😎", "Gen Z Teenage Drama"), ("🔥", "Steamy romance novel"), ("🤪", "Absurdist comedy"),
        ("🎭", "Melodramatic soap opera"), ("🎬", "High-budget action movie"), ("🤵", "Classic Bond film")
    ],
    "moods": [
        ("🍸", "Sexy and seductive"), ("💥", "Explosive and chaotic"), ("😂", "Ridiculously over-the-top"),
        ("😱", "Suspenseful and betrayal-filled"), ("🌟", "Glamorous and extravagant"),
        ("🥂", "Party-focused hedonism"), ("🔫", "Action-packed gunfights"), ("🕶️", "Cool and stylish")
    ]
}


# --- Helper functions ---

def get_story_options() -> Dict[str, List[Tuple[str, str]]]:
    return STORY_OPTIONS


def extract_field(data: Dict[str, Any], field: str, alt_field: Optional[str] = None, default: Any = None):
    return data.get(field) or data.get(alt_field) or default


def extract_character_traits(character_data: Dict[str, Any]) -> List[str]:
    return extract_field(character_data, 'character_traits', 'traits', [])


def extract_plot_lines(character_data: Dict[str, Any]) -> List[str]:
    return extract_field(character_data, 'plot_lines', 'plot_lines', [])


def extract_character_style(character_data: Dict[str, Any]) -> str:
    return extract_field(character_data, 'style', 'visual_description', '')


def extract_character_name(character_data: Dict[str, Any]) -> str:
    return extract_field(character_data, 'name', 'character_name', 'Unnamed Character')


def extract_character_role(character_data: Dict[str, Any]) -> str:
    return extract_field(character_data, 'role', 'character_role', 'neutral')


# --- Main story generation ---

def generate_story(
    conflict: str, setting: str, narrative_style: str, mood: str,
    character_info: Optional[Dict[str, Any]] = None,
    custom_conflict: Optional[str] = None, custom_setting: Optional[str] = None,
    custom_narrative: Optional[str] = None, custom_mood: Optional[str] = None,
    previous_choice: Optional[str] = None, story_context: Optional[str] = None,
    additional_characters: Optional[List[Dict[str, Any]]] = None,
    protagonist_name: Optional[str] = None, protagonist_gender: Optional[str] = None,
    protagonist_level: Optional[int] = 1
) -> Dict[str, Any]:

    final_conflict = custom_conflict or conflict
    final_setting = custom_setting or setting
    final_narrative = custom_narrative or narrative_style
    final_mood = custom_mood or mood

    protagonist_info = (
        f"You are {protagonist_name}, a {protagonist_gender} agent who is very charismatic and arrogant, constantly receiving romantic advances. "
        if protagonist_name and protagonist_gender else
        "You are a charismatic but reckless agent constantly receiving romantic advances. "
    )

    # Build Character and Additional Character prompts
    def format_character_info(char):
        traits = ', '.join(extract_character_traits(char))
        return f"- Name: {extract_character_name(char)}\n  Role: {extract_character_role(char)}\n  Traits: {traits}\n"

    selected_character_prompt = format_character_info(character_info) if character_info else ""
    additional_characters_prompt = "\n".join(format_character_info(char) for char in additional_characters or [])

    # Core Universe + Mission Requirements
    universe_prompt = (
        f"Conflict: {final_conflict}\nSetting: {final_setting}\nNarrative Style: {final_narrative}\nMood: {final_mood}\n\n"
        "This is a high-stakes, sexy, international world of espionage, luxury, and danger. "
        f"{protagonist_info} The story takes place in 2040 with advanced but recognizable technology.\n\n"
        "IMPORTANT: The story MUST begin with a mission assigned by a mission-giver (use a 'mission-giver' role character). "
        "The mission must include a clear objective, a specific target (villain role), a reward (💵, 💷, 💴, 💎), and a deadline.\n\n"
        "The mission must be fully structured as a JSON block alongside the story."
    )

    # Context from previous story, if any
    context_prompt = (
        f"Previous story context: {story_context}\nPlayer chose: {previous_choice}\n"
        if story_context and previous_choice else ""
    )

    # Final Prompt Assembly
    prompt = (
        f"{universe_prompt}\n\nCharacter Info:\n{selected_character_prompt}\nAdditional Characters:\n{additional_characters_prompt}\n\n"
        f"{context_prompt}\n"
        "Return a JSON object structured as:\n"
        "{\n"
        "  'title': 'Episode title',\n"
        "  'story': 'Narrative text',\n"
        "  'choices': [{'text': 'Choice text', 'consequence': 'Outcome hint', 'currency_requirements': {'💵': amount}}],\n"
        "  'mission': {'title': '...', 'description': '...', 'giver': '...', 'giver_id': '...', 'target': '...', 'target_id': '...', 'objective': '...', 'reward_currency': '...', 'reward_amount': '...', 'deadline': '...', 'difficulty': '...'},\n"
        "  'characters': ['List of character names featured']\n"
        "}"
    )

    # OpenAI call
    try:
        response = client.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.9,
            max_tokens=5000,
            response_format={"type": "json_object"}
        )
        result = json.loads(response.choices[0].message.content)
        return {
            "story": json.dumps(result),
            "conflict": final_conflict,
            "setting": final_setting,
            "narrative_style": final_narrative,
            "mood": final_mood
        }

    except Exception as e:
        logger.error(f"Error generating story: {str(e)}")
        raise Exception(f"Failed to generate story: {str(e)}")
