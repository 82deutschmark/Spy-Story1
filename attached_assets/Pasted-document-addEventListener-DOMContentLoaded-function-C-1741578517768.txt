  // ---------------------------
  // Helper Functions
  // ---------------------------

  // Show a Bootstrap toast notification
  function showNotification(title, message, type = 'info') {
    toastTitleEl.textContent = title;
    toastMessageEl.textContent = message;
    notificationToastEl.classList.remove('bg-success', 'bg-danger', 'bg-info');
    if (type === 'success') {
      notificationToastEl.classList.add('bg-success', 'text-white');
    } else if (type === 'error') {
      notificationToastEl.classList.add('bg-danger', 'text-white');
    } else {
      notificationToastEl.classList.add('bg-info', 'text-white');
    }
    new bootstrap.Toast(notificationToastEl).show();
  }
  // Parse analysis JSON from data attribute or content
  function parseAnalysis() {
    try {
      if (generatedContent.dataset.analysis) {
        return JSON.parse(generatedContent.dataset.analysis);
      } else if (generatedContent.textContent) {
        return JSON.parse(generatedContent.textContent);
      }
      throw new Error('No analysis data found');
    } catch (error) {
      showNotification('Error', 'Invalid JSON data: ' + error.message, 'error');
      return null;
    }
  }

  // Toggle form fields based on image type selection
  function toggleTypeFields() {
    if (imageType.value === 'character') {
      characterFields.style.display = 'block';
      sceneFields.style.display = 'none';
    } else {
      characterFields.style.display = 'none';
      sceneFields.style.display = 'block';
    }
  }

  // Populate the edit form with analysis data
  function populateEditForm() {
    const analysisData = parseAnalysis();
    if (!analysisData) return;
    // Infer type: if analysis.character exists, treat as character; otherwise scene.
    const isCharacter = analysisData.character && typeof analysisData.character === 'object' ||
                        (!analysisData.scene_type && !analysisData.setting);
    imageType.value = isCharacter ? 'character' : 'scene';
    if (isCharacter) {
      characterFields.style.display = 'block';
      sceneFields.style.display = 'none';
      const characterData = analysisData.character || {};
      imageName.value = characterData.code_name || analysisData.name || '';
      codeName.value = characterData.code_name || '';
      characterRole.value = characterData.role || 'undetermined';
      characterStyle.value = characterData.style || '';
      backstory.value = characterData.backstory || '';
      characterTraits.value = Array.isArray(characterData.character_traits) ? characterData.character_traits.join(', ') : '';
      plotLines.value = Array.isArray(characterData.plot_lines) ? characterData.plot_lines.join('\n') : '';
    } else {
      characterFields.style.display = 'none';
      sceneFields.style.display = 'block';
      imageName.value = analysisData.setting || '';
      sceneType.value = analysisData.scene_type || '';
      sceneSetting.value = analysisData.setting || '';
      dramaticMoments.value = Array.isArray(analysisData.dramatic_moments) ? analysisData.dramatic_moments.join('\n') : '';
    }
  }

  // Apply form edits to a clone of the original analysis object
  function applyEditsToAnalysis(originalAnalysis) {
    const analysis = JSON.parse(JSON.stringify(originalAnalysis));
    const nameVal = imageName.value;
    analysis.image_type = imageType.value;
    if (imageType.value === 'character') {
      analysis.name = nameVal;
      analysis.character = analysis.character || {};
      analysis.character.name = nameVal;
      analysis.character.code_name = codeName.value;
      analysis.character.role = characterRole.value;
      analysis.character.style = characterStyle.value;
      analysis.character.backstory = backstory.value;
      const traitsArr = characterTraits.value.split(',').map(t => t.trim()).filter(Boolean);
      analysis.character.character_traits = traitsArr;
      analysis.character_traits = traitsArr;
      const plotArr = plotLines.value.split('\n').map(p => p.trim()).filter(Boolean);
      analysis.character.plot_lines = plotArr;
      analysis.plot_lines = plotArr;
      // Remove any scene-related fields
      delete analysis.scene_type;
      delete analysis.setting;
      delete analysis.dramatic_moments;
    } else {
      analysis.name = nameVal;
      analysis.scene_type = sceneType.value;
      analysis.setting = sceneSetting.value;
      const momentsArr = dramaticMoments.value.split('\n').map(m => m.trim()).filter(Boolean);
      analysis.dramatic_moments = momentsArr;
      // Remove any character-related fields
      delete analysis.character;
      delete analysis.character_name;
      delete analysis.character_traits;
      delete analysis.role;
      delete analysis.plot_lines;
    }
    return analysis;
  }

  // Setup editing UI for analysis data
  function setupAnalysisEditing(analysis) {
    try {
      if (typeof analysis === 'string') {
        analysis = JSON.parse(analysis);
      }
      if (!analysis || typeof analysis !== 'object') return;
      populateEditForm();
      // Ensure toggle events are attached only once
      editModeSwitch.addEventListener('change', function() {
        editContainer.style.display = this.checked ? 'block' : 'none';
      });
      imageType.addEventListener('change', toggleTypeFields);
      applyChangesBtn.addEventListener('click', function() {
        try {
          const updatedAnalysis = applyEditsToAnalysis(analysis);
          generatedContent.textContent = JSON.stringify(updatedAnalysis, null, 2);
          generatedContent.dataset.analysis = JSON.stringify(updatedAnalysis);
          // Update our state with edited analysis
          analysis = updatedAnalysis;
          showNotification('Success', 'Changes applied to analysis', 'success');
        } catch (error) {
          showNotification('Error', 'Failed to apply changes: ' + error.message, 'error');
        }
      });
    } catch (error) {
      showNotification('Error', 'Failed to set up analysis editing: ' + error.message, 'error');
    }
  }

  // ---------------------------
  // Event Listeners
  // ---------------------------
  
  // Image analysis form submission
  if (imageForm) {
    imageForm.addEventListener('submit', function(e) {
      e.preventDefault();
      const imageUrl = imageUrlInput.value.trim();
      if (!imageUrl) {
        showNotification('Error', 'Please enter an image URL', 'error');
        return;
      }
      generatedContent.innerHTML = `<div class="text-center">
        <div class="spinner-border text-primary" role="status">
          <span class="visually-hidden">Loading...</span>
        </div>
        <p>Analyzing image...</p>
      </div>`;
      // Ensure result area is visible
      resultDiv.style.display = 'block';
      fetch('/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: new URLSearchParams({ 'image_url': imageUrl })
      })
      .then(response => response.json())
      .then(data => {
        if (data.success) {
          // Display the analysis JSON
          generatedContent.textContent = JSON.stringify(data.analysis, null, 2);
          generatedContent.dataset.analysis = JSON.stringify(data.analysis);
          generatedContent.dataset.imageUrl = data.image_url;
          currentImageData = {
            image_url: data.image_url,
            analysis: data.analysis,
            saved_to_db: data.saved_to_db || false
          };
          setupAnalysisEditing(data.analysis);
          if (saveToDbBtn) {
            saveToDbBtn.style.display = data.saved_to_db ? 'none' : 'inline-block';
          }
        } else {
          throw new Error(data.error || 'Failed to analyze image');
        }
      })
      .catch(error => {
        generatedContent.textContent = 'Error: ' + error.message;
        showNotification('Error', error.message, 'error');
      });
    });
  }

  // Copy analysis to clipboard
  if (copyBtn) {
    copyBtn.addEventListener('click', function() {
      navigator.clipboard.writeText(generatedContent.textContent)
      .then(() => showNotification('Success', 'Copied to clipboard', 'success'))
      .catch(err => showNotification('Error', 'Failed to copy: ' + err.message, 'error'));
    });
  }

  // Save analysis to database
  if (saveToDbBtn) {
    saveToDbBtn.addEventListener('click', function() {
      const analysis = parseAnalysis();
      if (!analysis) return;
      let imageUrl = generatedContent.dataset.imageUrl || (currentImageData && currentImageData.image_url);
      if (!imageUrl) {
        showNotification('Error', 'No image URL found', 'error');
        return;
      }
      this.disabled = true;
      this.innerHTML = `<span class="spinner-border spinner-border-sm" role="status" aria-hidden="true"></span> Saving...`;
      fetch('/save_analysis', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ image_url: imageUrl, analysis })
      })
      .then(response => response.json())
      .then(data => {
        if (data.error) {
          showNotification('Error', data.error, 'error');
          this.disabled = false;
          this.innerHTML = '<i class="fas fa-save me-2"></i>Save to Database';
          return;
        }
        showNotification('Success', 'Analysis saved to database', 'success');
        this.innerHTML = '<i class="fas fa-check me-2"></i>Saved';
        setTimeout(() => location.reload(), 2000);
      })
      .catch(error => {
        showNotification('Error', 'Failed to save analysis: ' + error.message, 'error');
        this.disabled = false;
        this.innerHTML = '<i class="fas fa-save me-2"></i>Save to Database';
      });
    });
  }

  if (editModeSwitch) {
    editModeSwitch.addEventListener('change', function() {
      editContainer.style.display = this.checked ? 'block' : 'none';
      if (this.checked) populateEditForm();

  function refreshImagesList() {
    if (!imagesTableBody) return;
    imagesTableBody.innerHTML = `<tr>
      <td colspan="6" class="text-center">
        <div class="spinner-border text-primary" role="status">
          <span class="visually-hidden">Loading...</span>
        </div>
      </td>
    </tr>`;
    fetch('/api/images/all?per_page=10')
    .then(response => response.json())
    .then(data => {
      if (data.error) {
        showNotification('Error', data.error, 'error');
        return;
      }
      if (data.images.length === 0) {
        imagesTableBody.innerHTML = `<tr>
          <td colspan="6" class="text-center">No image records found</td>
        </tr>`;
        return;
      }
      imagesTableBody.innerHTML = '';
      data.images.forEach(img => {
        imagesTableBody.innerHTML += `
          <tr data-id="${img.id}">
            <td>${img.id}</td>
            <td><img src="${img.image_url}" class="img-thumbnail" width="100" alt="Thumbnail"></td>
            <td>${img.image_type}</td>
            <td>${img.name || 'N/A'}</td>
            <td>${img.created_at}</td>
            <td>
              <div class="btn-group">
                <button class="btn btn-sm btn-info view-details-btn" data-id="${img.id}" title="View Details">
                  <i class="fas fa-eye"></i>
                </button>
                <button class="btn btn-sm btn-danger delete-image-btn" data-id="${img.id}" title="Delete Record">
                  <i class="fas fa-trash"></i>
                </button>
              </div>
            </td>
          </tr>`;
      });
    })
    .catch(error => showNotification('Error', 'Failed to load images: ' + error.message, 'error'));
  }

  function loadAllImages(page = 1, filter = '', search = '') {
    if (!allImagesTableBody) return;
    currentImagePage = page;
    currentImageFilter = filter;
    currentImageSearch = search;
    allImagesTableBody.innerHTML = `<tr>
      <td colspan="6" class="text-center">
        <div class="spinner-border text-primary" role="status">
          <span class="visually-hidden">Loading...</span>
        </div>
      </td>
    </tr>`;
    let url = `/api/images/all?page=${page}&per_page=20`;
    if (filter) url += `&type=${filter}`;
    if (search) url += `&search=${encodeURIComponent(search)}`;
    fetch(url)
    .then(response => response.json())
    .then(data => {
      if (data.error) {
        showNotification('Error', data.error, 'error');
        return;
      }
      if (data.images.length === 0) {
        allImagesTableBody.innerHTML = `<tr>
          <td colspan="6" class="text-center">No image records found</td>
        </tr>`;
        imagesPagination.innerHTML = '';
        return;
      }
      allImagesTableBody.innerHTML = '';
      data.images.forEach(img => {
        allImagesTableBody.innerHTML += `
          <tr data-id="${img.id}">
            <td>${img.id}</td>
            <td><img src="${img.image_url}" class="img-thumbnail" width="100" alt="Thumbnail"></td>
            <td>${img.image_type}</td>
            <td>${img.name || 'N/A'}</td>
            <td>${img.created_at}</td>
            <td>
              <div class="btn-group">
                <button class="btn btn-sm btn-info view-details-btn" data-id="${img.id}" title="View Details">
                  <i class="fas fa-eye"></i>
                </button>
                <button class="btn btn-sm btn-danger delete-image-btn" data-id="${img.id}" title="Delete Record">
                  <i class="fas fa-trash"></i>
                </button>
              </div>
            </td>
          </tr>`;
      });
      createImagePagination(data.pagination);
    })
    .catch(error => {
      allImagesTableBody.innerHTML = `<tr>
        <td colspan="6" class="text-center text-danger">Error loading images: ${error.message}</td>
      </tr>`;
      imagesPagination.innerHTML = '';
    });
  }

  function createImagePagination(pagination) {
    if (!imagesPagination) return;
    imagesPagination.innerHTML = '';
    // Previous button
    const prevItem = document.createElement('li');
    prevItem.className = `page-item ${pagination.page <= 1 ? 'disabled' : ''}`;
    const prevLink = document.createElement('a');
    prevLink.className = 'page-link';
    prevLink.href = '#';
    prevLink.textContent = 'Previous';
    if (pagination.page > 1) {
      prevLink.addEventListener('click', (e) => {
        e.preventDefault();
        loadAllImages(pagination.page - 1, currentImageFilter, currentImageSearch);
      });
    }
    prevItem.appendChild(prevLink);
    imagesPagination.appendChild(prevItem);
    // Page numbers
    const startPage = Math.max(1, pagination.page - 2);
    const endPage = Math.min(pagination.pages, pagination.page + 2);
    for (let i = startPage; i <= endPage; i++) {
      const pageItem = document.createElement('li');
      pageItem.className = `page-item ${i === pagination.page ? 'active' : ''}`;
      const pageLink = document.createElement('a');
      pageLink.className = 'page-link';
      pageLink.href = '#';
      pageLink.textContent = i;
      if (i !== pagination.page) {
        pageLink.addEventListener('click', (e) => {
          e.preventDefault();
          loadAllImages(i, currentImageFilter, currentImageSearch);
        });
      }
      pageItem.appendChild(pageLink);
      imagesPagination.appendChild(pageItem);
    }
    // Next button
    const nextItem = document.createElement('li');
    nextItem.className = `page-item ${pagination.page >= pagination.pages ? 'disabled' : ''}`;
    const nextLink = document.createElement('a');
    nextLink.className = 'page-link';
    nextLink.href = '#';
    nextLink.textContent = 'Next';
    if (pagination.page < pagination.pages) {
      nextLink.addEventListener('click', (e) => {
        e.preventDefault();
        loadAllImages(pagination.page + 1, currentImageFilter, currentImageSearch);
      });
    }
    nextItem.appendChild(nextLink);
    imagesPagination.appendChild(nextItem);
  }

  function loadStoryNodes() {
    if (!storyNodesTableBody) return;
    storyNodesTableBody.innerHTML = `<tr>
      <td colspan="6" class="text-center">
        <div class="spinner-border text-primary" role="status">
          <span class="visually-hidden">Loading...</span>
        </div>
      </td>
    </tr>`;
    fetch('/api/story_nodes/all')
    .then(response => response.json())
    .then(data => {
      if (data.error) {
        showNotification('Error', data.error, 'error');
        return;
      }
      if (data.nodes.length === 0) {
        storyNodesTableBody.innerHTML = `<tr>
          <td colspan="6" class="text-center">No story nodes found</td>
        </tr>`;
        return;
      }
      storyNodesTableBody.innerHTML = '';
      data.nodes.forEach(node => {
        storyNodesTableBody.innerHTML += `
          <tr data-id="${node.id}">
            <td>${node.id}</td>
            <td>${node.parent_node_id || 'None'}</td>
            <td>${node.text_preview}</td>
            <td>${node.choices_count}</td>
            <td>${node.is_endpoint ? 'Yes' : 'No'}</td>
            <td>
              <div class="btn-group">
                <button class="btn btn-sm btn-info view-node-btn" data-id="${node.id}" title="View Node">
                  <i class="fas fa-eye"></i>
                </button>
              </div>
            </td>
          </tr>`;
      });
    })
    .catch(error => {
      storyNodesTableBody.innerHTML = `<tr>
        <td colspan="6" class="text-center text-danger">Error loading story nodes: ${error.message}</td>
      </tr>`;
    });
  }

  // ---------------------------
  // Delegated Event Listeners for Details and Deletion
  // ---------------------------
  
  document.addEventListener('click', function(e) {
    // View details
    const viewBtn = e.target.closest('.view-details-btn');
    if (viewBtn) {
      const imageId = viewBtn.getAttribute('data-id');
      fetch(`/api/image/${imageId}`)
      .then(response => response.json())
      .then(data => {
        if (data.error) {
          showNotification('Error', data.error, 'error');
          return;
        }
        modalImage.src = data.image_url;
        modalContent.textContent = JSON.stringify(data.analysis, null, 2);
        currentImageData = {
          image_id: data.id,
          image_url: data.image_url,
          analysis: data.analysis
        };
        // Reset modal edit mode
        document.getElementById('editModeSwitch').checked = false;
        saveAnalysisBtn.style.display = 'none';
        new bootstrap.Modal(detailsModalEl).show();
      })
      .catch(error => showNotification('Error', 'Failed to load image details: ' + error.message, 'error'));
    }
    // Delete image
    const delImgBtn = e.target.closest('.delete-image-btn');
    if (delImgBtn) {
      const imageId = delImgBtn.getAttribute('data-id');
      if (confirm('Are you sure you want to delete this image record?')) {
        fetch(`/api/image/${imageId}`, { method: 'DELETE' })
        .then(response => response.json())
        .then(data => {
          if (data.error) {
            showNotification('Error', data.error, 'error');
            return;
          }
          showNotification('Success', data.message, 'success');
          const row = delImgBtn.closest('tr');
          if (row) row.remove();
          refreshImagesList();
          loadAllImages(currentImagePage, currentImageFilter, currentImageSearch);
        })
        .catch(error => showNotification('Error', 'Failed to delete image: ' + error.message, 'error'));
      }
    }
    // Delete story
    const delStoryBtn = e.target.closest('.delete-story-btn');
    if (delStoryBtn) {
      const storyId = delStoryBtn.getAttribute('data-id');
      if (confirm('Are you sure you want to delete this story record?')) {
        fetch(`/api/story/${storyId}`, { method: 'DELETE' })
        .then(response => response.json())
        .then(data => {
          if (data.error) {
            showNotification('Error', data.error, 'error');
            return;
          }
          showNotification('Success', data.message, 'success');
          const row = delStoryBtn.closest('tr');
          if (row) row.remove();
          refreshStoriesList();
          loadAllStories(currentStoryPage, currentStorySearch);
        })
        .catch(error => showNotification('Error', 'Failed to delete story: ' + error.message, 'error'));
      }
    }
  });

  // Delete all images and stories
  if (deleteAllImagesBtn) {
    deleteAllImagesBtn.addEventListener('click', function() {
      if (confirm('WARNING: Are you sure you want to delete ALL image records? This action cannot be undone.')) {
        fetch('/api/db/delete-all-images', { method: 'POST' })
        .then(response => response.json())
        .then(data => {
          if (data.error) {
            showNotification('Error', data.error, 'error');
            return;
          }
          showNotification('Success', data.message, 'success');
          if (imagesTableBody) imagesTableBody.innerHTML = '';
          refreshImagesList();
          loadAllImages();
        })
        .catch(error => showNotification('Error', 'Failed to delete all images: ' + error.message, 'error'));
      }
    });
  }
  if (deleteAllStoriesBtn) {
    deleteAllStoriesBtn.addEventListener('click', function() {
      if (confirm('WARNING: Are you sure you want to delete ALL story records? This action cannot be undone.')) {
        fetch('/api/db/delete-all-stories', { method: 'POST' })
        .then(response => response.json())
        .then(data => {
          if (data.error) {
            showNotification('Error', data.error, 'error');
            return;
          }
          showNotification('Success', data.message, 'success');
          if (storiesTableBody) storiesTableBody.innerHTML = '';
          refreshStoriesList();
          loadAllStories();
        })
        .catch(error => showNotification('Error', 'Failed to delete all stories: ' + error.message, 'error'));
      }
    });
  }

  // Health check
  if (runHealthCheckBtn) {
    runHealthCheckBtn.addEventListener('click', function() {
      runHealthCheckBtn.disabled = true;
      runHealthCheckBtn.innerHTML = `<span class="spinner-border spinner-border-sm" role="status" aria-hidden="true"></span> Running...`;
      fetch('/api/db/health-check')
      .then(response => response.json())
      .then(data => {
        runHealthCheckBtn.disabled = false;
        runHealthCheckBtn.innerHTML = '<i class="fas fa-stethoscope me-1"></i>Run Health Check';
        if (data.error) {
          showNotification('Error', data.error, 'error');
          return;
        }
        document.getElementById('totalImages').textContent = data.stats.image_count;
        document.getElementById('characterImages').textContent = data.stats.character_count;
        document.getElementById('sceneImages').textContent = data.stats.scene_count;
        document.getElementById('totalStories').textContent = data.stats.story_count;
        document.getElementById('orphanedImages').textContent = data.stats.orphaned_images;
        document.getElementById('emptyStories').textContent = data.stats.empty_stories;
        const noIssuesAlert = document.getElementById('noIssuesAlert');
        const issuesList = document.getElementById('issuesList');
        if (data.has_issues) {
          noIssuesAlert.style.display = 'block';
          issuesList.style.display = 'block';
          issuesList.innerHTML = '';
          data.issues.forEach(issue => {
            const severityClass = issue.severity === 'error' ? 'danger' : 'warning';
            issuesList.innerHTML += `<li class="list-group-item list-group-item-${severityClass}">
              <i class="fas fa-exclamation-triangle me-2"></i>${issue.message}
            </li>`;
          });
        } else {
          noIssuesAlert.style.display = 'block';
          issuesList.style.display = 'none';
        }
        showNotification('Success', 'Health check completed', 'success');
      })
      .catch(error => {
        runHealthCheckBtn.disabled = false;
        runHealthCheckBtn.innerHTML = '<i class="fas fa-stethoscope me-1"></i>Run Health Check';
        showNotification('Error', 'Failed to run health check: ' + error.message, 'error');
      });
    });
  }

  // Advanced image/story browser: search and filter
  if (imageSearchBtn && imageSearchInput) {
    imageSearchBtn.addEventListener('click', () => {
      loadAllImages(1, currentImageFilter, imageSearchInput.value.trim());
    });
    imageSearchInput.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') loadAllImages(1, currentImageFilter, imageSearchInput.value.trim());
    });
  }
  if (filterButtons) {
    filterButtons.forEach(button => {
      button.addEventListener('click', () => {
        const filter = button.getAttribute('data-filter');
        loadAllImages(1, filter, currentImageSearch);
      });
    });
  }
  if (storySearchBtn && storySearchInput) {
    storySearchBtn.addEventListener('click', () => {
      loadAllStories(1, storySearchInput.value.trim());
    });
    storySearchInput.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') loadAllStories(1, storySearchInput.value.trim());
    });
  }
  if (loadAllImagesBtn) {
    loadAllImagesBtn.addEventListener('click', () => loadAllImages(1, currentImageFilter, currentImageSearch));
  }
  if (loadNodesBtn) {
    loadNodesBtn.addEventListener('click', loadStoryNodes);
  }

  // ---------------------------
  // Reanalyze image (modal confirmation)
  // ---------------------------
  if (reanalyzeImageBtn) {
    reanalyzeImageBtn.addEventListener('click', function() {
      if (!currentImageData || !currentImageData.image_id) {
        showNotification('Error', 'No image selected for reanalysis', 'error');
        return;
      }
      new bootstrap.Modal(reanalyzeConfirmModalEl).show();
    });
  }
  if (confirmReanalyzeBtn) {
    confirmReanalyzeBtn.addEventListener('click', function() {
      new bootstrap.Modal(reanalyzeConfirmModalEl).hide();
      if (!currentImageData || !currentImageData.image_id) {
        showNotification('Error', 'No image selected for reanalysis', 'error');
        return;
      }
      confirmReanalyzeBtn.disabled = true;
      confirmReanalyzeBtn.innerHTML = `<span class="spinner-border spinner-border-sm" role="status" aria-hidden="true"></span> Processing...`;
      const preserveRelations = preserveRelationsCheck.checked;
      fetch(`/api/reanalyze/${currentImageData.image_id}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ preserve_relations: preserveRelations })
      })
      .then(response => response.json())
      .then(data => {
        confirmReanalyzeBtn.disabled = false;
        confirmReanalyzeBtn.innerHTML = '<i class="fas fa-sync-alt me-2"></i>Reanalyze';
        if (data.error) {
          showNotification('Error', data.error, 'error');
          return;
        }
        modalContent.textContent = JSON.stringify(data.analysis, null, 2);
        currentImageData.analysis = data.analysis;
        showNotification('Success', 'Image reanalyzed successfully', 'success');
        refreshImagesList();
        loadAllImages(currentImagePage, currentImageFilter, currentImageSearch);
      })
      .catch(error => {
        confirmReanalyzeBtn.disabled = false;
        confirmReanalyzeBtn.innerHTML = '<i class="fas fa-sync-alt me-2"></i>Reanalyze';
        showNotification('Error', 'Failed to reanalyze image: ' + error.message, 'error');
      });
    });
  }

  // ---------------------------
  // Initial Data Load
  // ---------------------------
  if (imagesTableBody) refreshImagesList();
  if (storiesTableBody) refreshStoriesList();
  if (allImagesTableBody) loadAllImages();
  if (allStoriesTableBody) loadAllStories();
  if (storyNodesTableBody) loadStoryNodes();

  // Delegated events for delete buttons in details view, etc. (see above)
});
